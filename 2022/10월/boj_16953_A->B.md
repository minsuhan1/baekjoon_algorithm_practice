### 코드
```python
import sys
input = sys.stdin.readline

a, b = map(int, input().split())

ans = 0
flag = False
while b >= a:
    if b == a:
        flag = True
        ans += 1
        print(ans)
        break
        
    nl = list(str(b))
    end = nl[-1]    
    
    if b % 2 == 0:
        b //= 2
    elif end == '1':
        nl.pop(-1)
        b = int(''.join(nl))
    else:
        break

    ans += 1

if not flag:
    print(-1)
```
